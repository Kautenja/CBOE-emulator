190  inline void fill(Side side, Quantity quantity, Price price) {
191  switch (side) {
192  case Side::Sell: {
193  shares -= quantity;
194  capital += quantity * price;
195  break;
196  }
197  case Side::Buy: {
198  shares += quantity;
199  capital -= quantity * price;
200  break;
201  }
202  }
203  }
