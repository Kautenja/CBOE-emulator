85 template<>
86 inline void find_best<Side::Buy>(Limit** highest_buy) {
87  if ((*highest_buy)->parent == nullptr) // removing root node
88  // set left because if right existed it would be a better limit already
89  *highest_buy = static_cast<Limit*>((*highest_buy)->left);
90  else // removing non-root node
91  *highest_buy = static_cast<Limit*>((*highest_buy)->parent);
92  if (*highest_buy != nullptr) // highest buy exists
93  *highest_buy = static_cast<Limit*>(BST::max(*highest_buy));
94 }

100 template<>
101 inline void find_best<Side::Sell>(Limit** lowest_sell) {
102  if ((*lowest_sell)->parent == nullptr) // removing root node
103  // set right because if left existed it would be a better limit already
104  *lowest_sell = static_cast<Limit*>((*lowest_sell)->right);
105  else // removing non-root node
106  *lowest_sell = static_cast<Limit*>((*lowest_sell)->parent);
107  if (*lowest_sell != nullptr) // lowest sell exists
108  *lowest_sell = static_cast<Limit*>(BST::min(*lowest_sell));
109 }

131 template<>
132 inline bool can_match<Side::Buy>(Price limit, Price market) {
133  // short circuit on price > 0: 0 is sentinel for no limit
134  // the price should be less than or equal to the best on the buy side.
135  // i.e., the submitted order is a sell limit with a lower price
136  return market == 0 || market <= limit;
137 }
138 
145 template<>
146 inline bool can_match<Side::Sell>(Price limit, Price market) {
147  // short circuit on price > 0: 0 is sentinel for no limit
148  // the price should be greater than or equal to the best on the sell side.
149  // i.e., the submitted order is a buy limit with a higher price
150  return market == 0 || market >= limit;
151 }

154 template<Side side>
155 struct LimitTree {
157  PriceLimitMap limits;
159  Limit* root = nullptr;
161  Limit* best = nullptr;
163  Count count = 0;
165  Volume volume = 0;
167  DataFeed::Sender* feed = nullptr;
169  static constexpr DataFeed::Side data_feed_side = DataFeed::bool_to_side(static_cast<bool>(side));
171  static constexpr DataFeed::Side data_feed_side_market = DataFeed::bool_to_side(!static_cast<bool>(side));
172 
177  LimitTree(DataFeed::Sender* feed_ = nullptr) : feed(feed_) { }
178 
180  void clear() {
181  // delete all the limits
182  for(auto item = limits.begin(); item != limits.end(); item++)
183  delete item->second;
184  // clear the pairs from the map
185  limits.clear();
186  // set remaining tracers to 0
187  root = nullptr;
188  best = nullptr;
189  count = 0;
190  volume = 0;
191  }

197  void limit(Order* order) {
198  if (limits.count(order->price) == 0) { // first order at limit price
199  // create the limit and set the order's limit to the limit
200  order->limit = new Limit(order);
201  // the limit price is new, add it to the limit tree
202  BST::insert(
203  reinterpret_cast<BST::Node<Price>**>(&root),
204  static_cast<BST::Node<Price>*>(order->limit)
205  );
206  // set the best price
207  set_best<side>(&best, order->limit);
208  // put the limit into the price to limit mapping
209  limits.emplace(order->limit->key, order->limit);
210  } else { // push to existing queue of orders at limit price
211  // set the order's limit to the limit node for the order's price
212  order->limit = limits.at(order->price);
213  // increment the count of the orders at this limit node
214  ++order->limit->count;
215  // increment the volume of the orders at this limit node
216  order->limit->volume += order->quantity;
217  // add the order to the queue of orders for the limit node
218  DLL::append(
219  reinterpret_cast<DLL::Node**>(&order->limit->order_tail),
220  static_cast<DLL::Node*>(order)
221  );
222  }
223  // update the count and volume for the entire tree
224  ++count;
225  volume += order->quantity;
226  // update the account
227  order->account->limit(order);
228  // send the quote to the feed
229  if (feed != nullptr)
230  feed->send<DataFeed::Messages::AddOrder>(order->uid, order->price, order->quantity, data_feed_side);
231  }
232 
233  // TODO: use object pool for limit reuse?
238  void remove(Order* order) {
239  // unwrap the limit for the order
240  auto limit_ = order->limit;
241  if (order->prev == nullptr && order->next == nullptr) { // last Order
242  // remove the limit from the tree
243  BST::remove(
244  reinterpret_cast<BST::Node<Price>**>(&root),
245  static_cast<BST::Node<Price>*>(limit_)
246  );
247  // replace the best limit if necessary
248  if (best == limit_)
249  find_best<side>(&best);
250  // remove the limit from the price map
251  limits.erase(limit_->key);
252  delete limit_;
253  } else { // at least 1 other Order in queue
254  --limit_->count;
255  limit_->volume -= order->quantity;
256  // remove the order from the queue (update head and tail)
257  DLL::remove(
258  reinterpret_cast<DLL::Node**>(&limit_->order_head),
259  reinterpret_cast<DLL::Node**>(&limit_->order_tail),
260  static_cast<DLL::Node*>(order)
261  );
262  }
263  // update the count and volume for the entire tree
264  --count;
265  volume -= order->quantity;
266  }

272  inline void cancel(Order* order) {
273  remove(order);
274  order->account->cancel(order);
275  // send the cancellation to the feed
276  if (feed != nullptr)
277  feed->send<DataFeed::Messages::DeleteOrder>(order->uid);

286  template<typename Callback>
287  void market(Order* order, Callback did_fill) {
288  // find orders until there are none
289  while (best != nullptr && can_match<side>(best->key, order->price)) {
290  // get the next match as the front of the best price
291  auto match = best->order_head;
292  if (match->quantity >= order->quantity) { // current match can fill
293  // update the account for the market order
294  order->account->market_fill(match, order);
295  if (match->quantity == order->quantity) { // limit order filled
296  // update the account for the limit order
297  match->account->limit_fill(match, order);
298  // send trade to the feed
299  if (feed != nullptr)
300  feed->send<DataFeed::Messages::Trade>(match->uid, match->price, match->quantity, data_feed_side_market);
301  // remove the current match from the book
302  remove(match);
303  did_fill(match->uid);
304  } else { // limit order partially filled
305  // remove the market order quantity from the limit quantity
306  match->quantity -= order->quantity;
307  // update the match's limit volume
308  match->limit->volume -= order->quantity;
309  // update the volume for the entire tree
310  volume -= order->quantity;
311  // update the account for the limit order
312  match->account->limit_partial(match, order);
313  // send trade to the feed
314  if (feed != nullptr)
315  feed->send<DataFeed::Messages::Trade>(match->uid, match->price, order->quantity, data_feed_side_market);
316  }
317  // set the order quantity to 0 since it's filled
318  order->quantity = 0;
319  return;
320  } // else: current match can NOT fill
321  // decrement the remaining quantity of the market order
322  order->quantity -= match->quantity;
323  // update the accounts for the orders
324  order->account->market_partial(match, order);
325  match->account->limit_fill(match, order);
326  // send trade to the feed
327  if (feed != nullptr)
328  feed->send<DataFeed::Messages::Trade>(match->uid, match->price, match->quantity, data_feed_side_market);
329  // remove the current match from the book
330  remove(match);
331  did_fill(match->uid);
332  }
333  }
