76  inline UID limit_sell(Account* account, Quantity quantity, Price price) {
77  // put the order into the sequence map
78  orders.emplace(std::piecewise_construct,
79  std::forward_as_tuple(sequence),
80  std::forward_as_tuple(sequence, Side::Sell, quantity, price, nullptr, account)
81  );
82  if (buys.best != nullptr && price <= buys.best->key) { // crosses
83  // place a market order with the limit price
84  buys.market(&orders.at(sequence), [&](UID uid) { orders.erase(uid); });
85  if (orders.at(sequence).quantity == 0) { // order filled
86  orders.erase(sequence);
87  return 0;
88  }
89  }
90  sells.limit(&orders.at(sequence));
91  return sequence++;
92  }

159  inline void cancel(UID order_id) {
160  auto order = &orders.at(order_id);
161  switch (order->side) { // remove the order from the appropriate side
162  case Side::Sell: { sells.cancel(order); break; }

173  inline void reduce(UID order_id, Quantity quantity) {
174  auto order = &orders.at(order_id);
175  if (quantity > order->quantity) { // trying to remove more than available
176  throw "trying to remove " +
177  std::to_string(quantity) +
178  " from order with " +
179  std::to_string(order->quantity) +
180  " available!";
181  }
182  // remove the quantity from the order and limit
183  order->quantity -= quantity;
184  order->limit->volume -= quantity;
185  switch (order->side) { // remove the order from the appropriate side
186  case Side::Sell: { sells.volume -= quantity; break; }
187  case Side::Buy: { buys.volume -= quantity; break; }
188  }
189  if (order->quantity == 0) { // remove the order if there is no quantity left
190  switch (order->side) { // remove the order from the appropriate side
191  case Side::Sell: { sells.cancel(order); break; }
192  case Side::Buy: { buys.cancel(order); break; }
193  }
194  orders.erase(order_id);
195  }
196  }

326  inline bool does_cross(const LimitOrderBook& other, Quantity spread = 0) const {
327  // check if A has a sell price
328  if (best_sell() == 0)
329  return false;
330  // check for overflow
331  if (best_sell() > std::numeric_limits<Price>::max() - spread)
332  return false;
333  // check if A is selling for less than B is buying
334  if (best_sell() + spread < other.best_buy())
335  return true;
336  return false;
337  }
