68  inline void limit_sell(UID order_id, Quantity quantity, Price price) {
69  // put the order into the map
70  orders.emplace(std::piecewise_construct,
71  std::forward_as_tuple(order_id),
72  std::forward_as_tuple(order_id, Side::Sell, quantity, price)
73  );
74  if (buys.best != nullptr && price <= buys.best->key) { // crosses
75  // place a market order with the limit price
76  buys.market(&orders.at(order_id), [&](UID uid) { orders.erase(uid); });
77  if (orders.at(order_id).quantity == 0) { // order filled
78  orders.erase(order_id);
79  return;
80  }
81  }
82  sells.limit(&orders.at(order_id));
83  }

92  inline void limit_buy(UID order_id, Quantity quantity, Price price) {
93  // put the order into the map
94  orders.emplace(std::piecewise_construct,
95  std::forward_as_tuple(order_id),
96  std::forward_as_tuple(order_id, Side::Buy, quantity, price)
97  );
98  if (sells.best != nullptr && price >= sells.best->key) { // crosses
99  // place a market order with the limit price
100  sells.market(&orders.at(order_id), [&](UID uid) { orders.erase(uid); });
101  if (orders.at(order_id).quantity == 0) { // order filled
102  orders.erase(order_id);
103  return;
104  }
105  }

107  }

142  inline void cancel(UID order_id) {
143  auto order = &orders.at(order_id);
144  switch (order->side) { // remove the order from the appropriate side
145  case Side::Sell: { sells.cancel(order); break; }

156  inline void reduce(UID order_id, Quantity quantity) {
157  auto order = &orders.at(order_id);
158  if (quantity > order->quantity) { // trying to remove more than available
159  throw "trying to remove " +
160  std::to_string(quantity) +
161  " from order with " +
162  std::to_string(order->quantity) +
163  " available!";
164  }
165  // remove the quantity from the order and limit
166  order->quantity -= quantity;
167  order->limit->volume -= quantity;
168  switch (order->side) { // remove the order from the appropriate side
169  case Side::Sell: { sells.volume -= quantity; break; }
170  case Side::Buy: { buys.volume -= quantity; break; }
171  }
172  if (order->quantity == 0) { // remove the order if there is no quantity left
173  switch (order->side) { // remove the order from the appropriate side
174  case Side::Sell: { sells.cancel(order); break; }
175  case Side::Buy: { buys.cancel(order); break; }
176  }
177  orders.erase(order_id);
178  }
179  }

247  inline Price price() const {
248  if (sells.best == nullptr && buys.best == nullptr) // empty book
249  return 0;
250  if (sells.best == nullptr) // no sell orders
251  return buys.best->key;
252  if (buys.best == nullptr) // no buy orders
253  return sells.best->key;
254  // return the midpoint
255  return (sells.best->key + buys.best->key) / 2;
256  }

360  inline bool does_cross(const LimitOrderBook& other, Quantity spread = 0) const {
361  // check if A has a sell price
362  if (best_sell() == 0)
363  return false;
364  // check for overflow
365  if (best_sell() > std::numeric_limits<Price>::max() - spread)
366  return false;
367  // check if A is selling for less than B is buying
368  if (best_sell() + spread < other.best_buy())
369  return true;
370  return false;
371  }
