73  inline void market_order(const std::string& line) {
74  std::smatch match;
75  std::regex_search(line, match, MARKET);
76  // determine the side of the order
77  auto side = bool_to_side(match.str(1).compare("sell"));
78  // determine the size of the order
79  Quantity size;
80  std::istringstream(match.str(2)) >> size;
81  // place the order
82  client.send<Messages::OrderRequest>(Messages::ORDER_PRICE_MARKET, size, side);
83  }

190  void parse(const std::string& line) {
191  if (std::regex_match(line, LOGIN))
192  login(line);
193  else if (std::regex_match(line, LOGOUT))
194  logout(line);
195  else if (std::regex_match(line, MARKET))
196  market_order(line);
197  else if (std::regex_match(line, LIMIT))
198  limit_order(line);
199  else if (std::regex_match(line, CANCEL))
200  cancel_order(line);
201  else if (std::regex_match(line, REPLACE))
202  replace_order(line);
203  else if (std::regex_match(line, PURGE))
204  purge_orders(line);
205  else if (std::regex_match(line, ACCOUNT))
206  print_account(line);
207  else
208  std::cout << "invalid command" << std::endl;
209  }
